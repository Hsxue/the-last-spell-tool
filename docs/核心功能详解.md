**批量更新优化**：

```typescript
// 使用 setTerrainBatch 进行高效批量更新
setTerrainBatch: (terrainMap: Map<string, TerrainType>) => {
  // 一次性创建新 Map，避免多次 React 重渲染
  const newTerrain = new Map(state.mapData.terrain);
  terrainMap.forEach((terrain, key) => {
    if (terrain === 'Empty') {
      newTerrain.delete(key);
    } else {
      newTerrain.set(key, terrain);
    }
  });
  state.mapData.terrain = newTerrain;
}
```

### 2.3 放置规则

**碰撞检测流程**：

```typescript
// 1. 获取建筑蓝图
const blueprint = getBlueprint(buildingId);

// 2. 计算实际占用格
const occupiedTiles = getOccupiedTiles(buildingId, positionX, positionY);
// 公式：baseX = positionX - blueprint.originX
//       baseY = positionY - blueprint.originY

// 3. 边界检查
for (const [x, y] of occupiedTiles) {
  if (x < 0 || x >= mapWidth || y < 0 || y >= mapHeight) {
    return { valid: false, reason: '超出边界' };
  }
}

// 4. 碰撞检查（O(1) Set 查找）
const occupiedSet = getAllOccupiedTiles(existingBuildings);
for (const [x, y] of occupiedTiles) {
  if (occupiedSet.has(`${x},${y}`)) {
    return { valid: false, reason: '与现有建筑重叠' };
  }
}

return { valid: true };
```

```typescript
// 只渲染可见区域 + 2 格安全边界
export function getVisibleTileRange(
  viewport: Viewport,
  mapWidth: number,
  mapHeight: number,
  tileSize: number
) {
  const margin = 2;
  return {
    minX: Math.max(0, Math.floor(viewport.x / tileSize) - margin),
    maxX: Math.min(mapWidth, Math.floor((viewport.x + viewport.width) / tileSize) + margin),
    // ...
  };
}
```

```typescript
// 根据 zoom 级别调整网格线密度
```
